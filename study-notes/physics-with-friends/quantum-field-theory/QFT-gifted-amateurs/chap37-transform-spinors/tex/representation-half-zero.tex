\begin{frame}{\headingColor\bf\Large Left-hand Weyl spinors, i.e., $\left(\frac{1}{2},0\right)$ rep'n of \,$\mathfrak{spin}(1,3)$}

\scriptsize
\vskip 5.0px

\textbf{Recall:}
\vskip 1px
\begin{itemize}
\item
	The finite-dimensional irreducible representations of \,$\SU(2)$\,
	(equivalently, those of $\mathfrak{su}(2)$, due to simple-connectedness of $\SU(2)$)
	have been classified.
	\vskip 2.5px
	For each \,$s = 0, \frac{1}{2}, 1, \frac{3}{2},\, \ldots\,$,\, there exists a unique representation 
	\,$\rho_{s} : \SU(2) \longrightarrow \GL(\Re,2s+1)$
	\vskip 5px
\item
	$\mathfrak{spin}(1,3) \,\cong\, \mathfrak{sl}(2,\C) \,\cong\, \mathfrak{su}(2) \oplus \i\,\mathfrak{su}(2)$\;
	(the copies of $\mathfrak{su}(2)$ commute)
	\vskip 1px
	{\tiny(Skew self-adjoint matrices with trace zero plus self-adjoint matrices with trace zero gives all matrices with trace zero.)}
	\vskip 5px
\item
	The irreducible representations of \,$\Spin^{\uparrow}(1,3)$\, are
	{\color{red}parametrized by the ordered pairs \,$(s_{+},s_{-})$\,} of non-negative multiples of \,$\frac{1}{2}$,\, where
	\,$s_{+}$\, refers to \,$\mathfrak{su}(2) \subset \mathfrak{sl}(2,\C)$\,
	and
	\,$s_{-}$\, refers to \,$\i\,\mathfrak{su}(2) \subset \mathfrak{sl}(2,\C)$.
	\vskip 2.5px
	See Theorem on page 517, \cite{Woit2017}.
\end{itemize}

\vskip 10px

\textbf{$(\frac{1}{2},0)$ representation of $\mathfrak{spin}(1,3)$, i.e., left-handed Weyl spinors}
\vskip 3px
\begin{itemize}
\item
	Generators of $\mathfrak{spin}(1,3)$:
	\vskip -15px
	\begin{equation*}
	N^{\pm}_{i} \,:=\, \frac{1}{2}\cdot\left(\,J_{i}\pm \i\,K_{i}\,\right),
	\quad
	\left[\,N^{+}_{i}\,,\,N^{+}_{j}\,\right] \, = \, \i\cdot\varepsilon_{ijk} \cdot N^{+}_{k},
	\quad
	\left[\,N^{-}_{i}\,,\,N^{-}_{j}\,\right] \, = \, \i\cdot\varepsilon_{ijk} \cdot N^{-}_{k},
	\quad
	\left[\,N^{-}_{i}\,,\,N^{+}_{j}\,\right] \,= \, 0
	\end{equation*}
\item
	The $(\frac{1}{2},0)$ representation:
	\vskip -5px
	\begin{equation*}
	\left\langle\,N^{+}_{x}, N^{+}_{y}, N^{+}_{z}\,\right\rangle \cong (s=\frac{1}{2}) \;\textnormal{rep'n of $\SU(2)$}\,,
	\quad\quad
	\left\langle\,N^{-}_{x}, N^{-}_{y}, N^{-}_{z}\,\right\rangle \cong (s=0) \;\textnormal{rep'n of $\SU(2)$}
	\end{equation*}
\end{itemize}

\end{frame}
\normalsize

\begin{frame}{\headingColor\bf\Large Left-handed Weyl spinors, i.e., $\left(\frac{1}{2},0\right)$ rep'n of \,$\mathfrak{spin}(1,3)$}

\scriptsize
\vskip 10.0px

\begin{itemize}
\item
	Generators of $\mathfrak{spin}(1,3)$:
	\vskip -0.3cm
	\begin{equation*}
	N^{\pm}_{i} \,:=\, \frac{1}{2}\cdot\left(\,J_{i}\pm \i\,K_{i}\,\right),
	\quad
	\left[\,N^{+}_{i}\,,\,N^{+}_{j}\,\right] \, = \, \i\cdot\varepsilon_{ijk} \cdot N^{+}_{k},
	\quad
	\left[\,N^{-}_{i}\,,\,N^{-}_{j}\,\right] \, = \, \i\cdot\varepsilon_{ijk} \cdot N^{-}_{k},
	\quad
	\left[\,N^{-}_{i}\,,\,N^{+}_{j}\,\right] \,= \, 0
	\end{equation*}
	
\item
	\vskip 0.1cm
	$(\frac{1}{2},0)$ rep'n:
	\;\;
	$\left\langle\,N^{+}_{x}, N^{+}_{y}, N^{+}_{z}\,\right\rangle \cong (s=\frac{1}{2}) \;\textnormal{rep'n of $\SU(2)$}$\,,
	\;\;
	$\left\langle\,N^{-}_{x}, N^{-}_{y}, N^{-}_{z}\,\right\rangle \cong (s=0) \;\textnormal{rep'n of $\SU(2)$}$
	\vskip 0.4cm
\item
	$\left\langle\,N^{-}_{x}, N^{-}_{y}, N^{-}_{z}\,\right\rangle \cong (s=0) \;\textnormal{rep'n of $\SU(2)$}$
	\;\;$\Longrightarrow$\;\;
	\vskip 1px
	\begin{equation*}
	\left(\rho_{(1/2,0)}\right)_{\!*}\!\left(\,N^{-}_{i}\,\right) \,=\, N^{-}_{i} \,:=\, \frac{1}{2}\cdot\left(\,J_{i}-\i\,K_{i}\,\right) \,=\, 0
	\quad\Longrightarrow\quad
	J_{i} \,=\, \i\,K_{i}\,,
	\end{equation*}
	where
	\;$\rho_{1/2,0} : \Spin^{\uparrow}(1,3) \longrightarrow \GL(\C^{2})$\; and
	\vskip -2px
	\begin{equation*}
	{\color{red}
	\left(\rho_{1/2,0}\right)_{\!*} :
		T_{I}\,\Spin^{\uparrow}(1,3) = \mathfrak{spin}(1,3)
		\longrightarrow
		T_{I}\,\GL(\C^{2}) = \textnormal{End}(\C^{2})
	}
	\end{equation*}
	\vskip 0.4cm
\item
	$\left\langle\,N^{+}_{x}, N^{+}_{y}, N^{+}_{z}\,\right\rangle \cong (s=\frac{1}{2})$
	\;\;$\Longrightarrow$\;\;
	\vskip 1px
	\begin{equation*}
	{\color{red}\dfrac{1}{2}\cdot\sigma_{i} \;=\; \left(\rho_{(1/2,0)}\right)_{\!*}\!\left(\,N^{+}_{i}\,\right) \;=\; N^{+}_{i}}
	\,:=\,
		\frac{1}{2}\cdot\left(\,J_{i} + \i\,K_{i}\,\right)
	\,=\,
		\frac{1}{2}\cdot\left(\,J_{i} + J_{i}\,\right)
	\,=\,
		{\color{red}J_{i}}
	\,=\,
		{\color{red}\i\,K_{i}}
	\end{equation*}
\end{itemize}

\end{frame}
\normalsize

\begin{frame}{\headingColor\bf\Large Left-handed Weyl spinors, i.e., $\left(\frac{1}{2},0\right)$ rep'n of \,$\mathfrak{spin}(1,3)$}

\scriptsize
\vskip 7.5px

\begin{itemize}
\item
	$\left\langle\,N^{+}_{x}, N^{+}_{y}, N^{+}_{z}\,\right\rangle \cong (s=\frac{1}{2})$
	\;\;$\Longrightarrow$\;\;
	${\color{red}\dfrac{1}{2}\cdot\sigma_{i} \;=\; N^{+}_{i}}
	\,=\,
		\frac{1}{2}\cdot\left(\,J_{i} + \i\,K_{i}\,\right)
	\,=\,
		\frac{1}{2}\cdot\left(\,J_{i} + J_{i}\,\right)
	\,=\,
		{\color{red}J_{i}}
	\,=\,
		{\color{red}\i\,K_{i}}
	$
	\vskip 5px
\item
	Rotations:\;\;
	${\color{red}R(\,\mathbf{\theta}\,)}
	\; = \;
		\exp\!\left(\,\overset{{\color{white}.}}{\i}\cdot(\theta_{x}J_{x}+\theta_{y}J_{y}+\theta_{z}J_{z})\,\right)
	\; = \;
		{\color{red}\exp\!\left(\,\frac{\sqrt{-1}}{2}\cdot(\theta_{x}\sigma_{x}+\theta_{y}\sigma_{y}+\theta_{z}\sigma_{z})\,\right)}
	$
	\vskip -7.5px
	\begin{eqnarray*}
	R_{x}(\theta_{x})\!\!
	& = &
		%\!\!\exp\!\left(\,\overset{{\color{white}.}}{\i}\cdot\theta_{x}J_{x}\,\right)
	%\; = \;
		\exp\!\left(\,\i\cdot\dfrac{\theta_{x}}{2}\cdot\sigma_{x}\,\right)
	\; = \;
		I_{2}
		+ \dfrac{\i\,\theta_{x}}{2}\left(\!\!\begin{array}{cc} 0 & 1 \\ 1 & 0 \end{array}\!\!\!\right)
		+ \dfrac{1}{2!}\!\left(\,\dfrac{\i\,\theta_{x}}{2}\left(\!\!\begin{array}{cc} 0 & 1 \\ 1 & 0 \end{array}\!\!\!\right)\right)^{2}
		+ \dfrac{1}{3!}\!\left(\,\dfrac{\i\,\theta_{x}}{2}\left(\!\!\begin{array}{cc} 0 & 1 \\ 1 & 0 \end{array}\!\!\!\right)\right)^{3}
		+ \;\cdots
	\\
	& = &
		{\color{blue}I_{2}}
		+ \i\cdot\!\left(\dfrac{\theta_{x}}{2}\right)\!\left(\!\!\begin{array}{cc} 0 & 1 \\ 1 & 0 \end{array}\!\!\!\right)
		+ {\color{blue}\dfrac{\i^{\,2}}{2!}\!\left(\!\dfrac{\theta_{x}}{2}\!\right)^{\!2}\left(\!\!\begin{array}{cc} 1 & 0 \\ 0 & 1 \end{array}\!\!\!\right)}
		+ \dfrac{\i^{\,3}}{3!}\!\left(\!\dfrac{\theta_{x}}{2}\!\right)^{\!3}\left(\!\!\begin{array}{cc} 0 & 1 \\ 1 & 0 \end{array}\!\!\!\right)
		+ {\color{blue}\dfrac{\i^{\,4}}{4!}\!\left(\!\dfrac{\theta_{x}}{2}\!\right)^{\!4}\left(\!\!\begin{array}{cc} 1 & 0 \\ 0 & 1 \end{array}\!\!\!\right)}
		+ \;\cdots
	\\
	& = &
		\left(\!\!\begin{array}{rr}
			{\color{blue}\cos(\theta_{x}/2)} & \i\,\sin(\theta_{x}/2)
			\\
			\i\,\sin(\theta_{x}/2) & {\color{blue}\cos(\theta_{x}/2)}
			\end{array}\!\!\!\right)
	\end{eqnarray*}
	\vskip 5px
\item
	Boosts:\;\;
	${\color{red}B(\,\mathbf{\phi}\,)}
	\; = \;
		\exp\!\left(\,\overset{{\color{white}.}}{\i}\cdot(\phi_{x}K_{x}+\phi_{y}K_{y}+\phi_{z}K_{z})\,\right)
	\; = \;
		{\color{red}\exp\!\left(\,\frac{1}{2}\cdot(\phi_{x}\sigma_{x}+\phi_{y}\sigma_{y}+\phi_{z}\sigma_{z})\,\right)}
		$
	\vskip -7.5px
	\begin{eqnarray*}
	B_{x}(\phi_{x})\!\!
	& = &
		%\!\!\exp\!\left(\,\phi_{x}\cdot\overset{{\color{white}.}}{\i}\,K_{x}\,\right)
	%\; = \;
		\exp\!\left(\,\dfrac{\phi_{x}}{2}\cdot\sigma_{x}\,\right)
	\; = \;
		I_{2}
		+ \dfrac{\phi_{x}}{2}\left(\!\!\begin{array}{cc} 0 & 1 \\ 1 & 0 \end{array}\!\!\!\right)
		+ \dfrac{1}{2}\left(\,\dfrac{\phi_{x}}{2}\left(\!\!\begin{array}{cc} 0 & 1 \\ 1 & 0 \end{array}\!\!\!\right)\right)^{2}
		+ \;\cdots
	\\
	& = &
		I_{2}
		+ \dfrac{\phi_{x}}{2}\left(\!\!\begin{array}{cc} 0 & 1 \\ 1 & 0 \end{array}\!\!\!\right)
		+ \dfrac{1}{2}\left(\!\dfrac{\phi_{x}}{2}\!\right)^{\!2}\left(\!\!\begin{array}{cc} 1 & 0 \\ 0 & 1 \end{array}\!\!\!\right)
		+ \dfrac{1}{3!}\left(\!\dfrac{\phi_{x}}{2}\!\right)^{\!3}\left(\!\!\begin{array}{cc} 0 & 1 \\ 1 & 0 \end{array}\!\!\!\right)
		+ \dfrac{1}{4!}\left(\!\dfrac{\phi_{x}}{2}\!\right)^{\!4}\left(\!\!\begin{array}{cc} 1 & 0 \\ 0 & 1 \end{array}\!\!\!\right)
		+ \;\cdots
	\\
	& = &
		\left(\!\!\begin{array}{rr} \cosh(\phi_{x}/2) & \sinh(\phi_{x}/2) \\ \sinh(\phi_{x}/2) & \cosh(\phi_{x}/2) \end{array}\!\!\!\right)
	\end{eqnarray*}
\end{itemize}

\end{frame}
\normalsize
