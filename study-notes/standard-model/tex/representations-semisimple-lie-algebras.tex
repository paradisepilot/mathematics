\section{Semisimplicity $\Longleftrightarrow$ direct sum of simple Lie algebras}

        %%%%% ~~~~~~~~~~~~~~~~~~~~ %%%%%

\vskip 0.5cm
\begin{definition}[Irreducibility and simplicity of Lie algebras]
\mbox{}
\vskip 0.1cm
\noindent
A Lie algebra \,$\mathfrak{g}$\, is said to be
\begin{itemize}
\item
    \textbf{irreducible} if the only ideals in \,$\mathfrak{g}$\,
    are \,$\mathfrak{g}$\, and \,$\{\,0\,\}$\,, and
\item
    \textbf{simple} if \,$\mathfrak{g}$\, is irreducible and
    \,{\color{red}$\dim(\,\mathfrak{g}\,) \,\geq\, 2$}.\,
\end{itemize}
\end{definition}

        %%%%% ~~~~~~~~~~~~~~~~~~~~ %%%%%

\begin{definition}[Reductiveness and semisimplicity of complex Lie algebras]
\mbox{}
\vskip 0.1cm
\noindent
A complex Lie algebra \,$\mathfrak{g}$\, is said to be
\begin{itemize}
\item
    \textbf{reductive} if there exists a real Lie subalgebra
    \,$\mathfrak{k} \,\subset\, \mathfrak{g}$\,
    such that
    \,$\mathfrak{g} \,\cong\, \mathfrak{k}_{\C}$,\,
    where \,$\mathfrak{k}$\, is isomorphic
    to the Lie algebra of a {\color{red}compact} matrix Lie group, and
\item
    \textbf{semisimple} if \,$\mathfrak{g}$\, is reductive and
    the centre of \,$\mathfrak{g}$\, is trivial.
\end{itemize}
\end{definition}

        %%%%% ~~~~~~~~~~~~~~~~~~~~ %%%%%

\begin{definition}[Compact real forms of semisimple complex Lie algebras]
\mbox{}
\vskip 0.1cm
\noindent
Suppose \,$\mathfrak{g}$\, is a semisimple complex Lie algebra.
A real Lie subalgebra \,$\mathfrak{k} \,\subset \mathfrak{g}\,$\,
is called a \textbf{compact real form} of \,$\mathfrak{g}$\, if
\,$\mathfrak{k}$\, is isomorphic to the Lie algebra of some
{\color{red}compact} matrix Lie group, and, for each \,$Z \in \mathfrak{g}$,\,
there exist unique \,$X,\, Y \in \mathfrak{k}$\, such that
\,$Z \,=\, X \,+\, \i\,Y$.\,
\end{definition}

        %%%%% ~~~~~~~~~~~~~~~~~~~~ %%%%%

The first key result we would like to establish is Theorem \ref{SemisimpleImpliesDirectSumOfSimples}.
Before stating and proving this theorem, we first establish a number of auxiliary results.

        %%%%% ~~~~~~~~~~~~~~~~~~~~ %%%%%

\vskip 0.5cm
\begin{proposition}[Reductive $\Longrightarrow$ ``unitary'' inner product exists on compact real form]
\label{ReductiveImpliesUnitaryInnerProductOnRealForm}
\mbox{}
\vskip 0.1cm
\noindent
Suppose \,$\mathfrak{g} \,=\, \mathfrak{k}_{\C}$\, is a reductive complex Lie algebra.
Then, the following statements are true:
\begin{enumerate}
\item
    There exists an $\Re$-valued inner product
    \,$\langle\,\cdot\,,\,\cdot\,\rangle : \mathfrak{k} \times \mathfrak{k} \longrightarrow \Re$\,
    on \,$\mathfrak{k}$\,
    with respect to which the adjoint action of
    \,$\mathfrak{k}$\, on \,$\mathfrak{g}$\,
    is ``unitary'' in the sense that
    \begin{equation}\label{almostUnitarity}
    \langle\,\ad_{X}(Y)\,,\,Z\,\rangle
    \;\; = \;\;
        -\,\langle\,Y\,,\,\ad_{X}(Z)\,\rangle\,,
    \quad
    \textnormal{for each \,$X \in \mathfrak{k}$\, and each \,$Y,Z \in \mathfrak{g}$}
    \end{equation}
\item
    If we define an operation
    \,$\mathfrak{g} \longrightarrow \mathfrak{g} : X \longmapsto X^{*}$\,
    by
    \begin{equation*}
    \left(\,\overset{{\color{white}.}}{X_{1} + \i\cdot X_{2}}\,\right)^{*}
    \;\; := \;\;
        -\,X_{1} + \i\cdot X_{2}\,,
    \quad
    \textnormal{for each \,$X_{1}, X_{2} \in \mathfrak{k}$}\,,
    \end{equation*}
    then any inner product on \,$\mathfrak{g}$\, satisfying \eqref{almostUnitarity}
    also satisfies:
    \begin{equation}
    \langle\,\ad_{X}(Y)\,,\,Z\,\rangle
    \;\; = \;\;
        \langle\,Y\,,\,\ad_{X^{*}}(Z)\,\rangle\,,
    \quad
    \textnormal{for each \,$X,Y,Z \in \mathfrak{g}$}
    \end{equation}
\end{enumerate}
\end{proposition}
\proof
\begin{enumerate}
\item
    By hypothesis, \,$\mathfrak{k}$\, is the Lie algebra of a compact (real) matrix Lie group \,$K$.\,
    Next, recall that every complex representation of a compact Lie group is
    unitary\footnote{If $G$ is a compact Lie group and $\pi : G \longrightarrow \GL(V)$
    is a finite-dimensional complex representation,
    then there exists a $G$-invariant Hermitian inner product
    $\langle\,\cdot\,,\,\cdot\,\rangle$ on $V$,
    i.e., $\langle\,\pi(g)\cdot v \,,\,\pi(g)\cdot w\,\rangle \,=\, \langle\,v\,,\,w\,\rangle$,\,
    for each $g \in G$, and each $v, w \in V$.}; see, for example, Lemma 5.6, p.100, \cite{Sepanski2010}.
    Hence, there exists an ($\Re$-valued) inner product
    \,$\langle\,\cdot\,,\,\cdot\,\rangle : \mathfrak{k} \times \mathfrak{k} \longrightarrow \Re$\,
    that is (a) invariant with respect to the adjoint action
    \,$\Ad : K \longrightarrow \GL(\mathfrak{k})$,\, and (b) skew-symmetric with respect to
    the infinitesimal adjoint
    action\footnote{(differential of $\Ad : K \longrightarrow \GL(\mathfrak{k})$)}
    \,$\ad : \mathfrak{k} \longrightarrow \gl(\mathfrak{k})$,\, i.e.,
    \begin{equation}
    \langle\,\Ad(k) \cdot Y\,,\,\Ad(k) \cdot Z\,\rangle 
    \;\; = \;\;
        \langle\,Y\,,\,Z\,\rangle\,,
    \quad
    \textnormal{for each \,$k \in K$\, and each \,$Y,Z \in \mathfrak{k}$}
    \end{equation}
    and
    \begin{equation}
    \langle\,\ad_{X}(Y)\,,\,Z\,\rangle \,+\, \langle\,Y\,,\,\ad_{X}(Z)\,\rangle
    \;\; = \;\;
        0\,,
    \quad
    \textnormal{for each \,$X,Y,Z \in \mathfrak{k}$}
    \end{equation}
    Extension of \,$\ad_{X}$\, by {\color{red}$\C$-linearity} and
    extension of \,$\langle\,\cdot\,,\,\cdot\,\rangle$\, by {\color{red}$\C$-bilinearity}
    (anti-linear in the first argument)
    together yield the desired equality.
    Indeed, for each \,$X \in \mathfrak{k}$\, and each \,$Y, Z \in \mathfrak{g}$,\,
    where \,$Y = Y_{1} + \i\,Y_{2}$\, and \,$Z = Z_{1} + \i\,Z_{2}$\,
    with \,$Y_{1}, Y_{2}, Z_{1}, Z_{2} \in \mathfrak{k}$,\, we have:
    \begin{eqnarray*}
    \langle\,\ad_{X}(Y)\,,\,Z\,\rangle
    & = &
        \langle\,\ad_{X}(Y_{1}+\i\,Y_{2})\,,\,Z_{1} + \i\,Z_{2}\,\rangle
    \;\; = \;\;
        \langle\,\ad_{X}(Y_{1})+\i\,\ad_{X}(Y_{2})\,,\,Z_{1} + \i\,Z_{2}\,\rangle
    \\
    & = &
        \langle\,\ad_{X}(Y_{1})\,,\,Z_{1}\,\rangle
        \,+\,
        \langle\,\ad_{X}(Y_{1})\,,\,\i\,Z_{2}\,\rangle
        \,+\,
        \langle\,\i\,\ad_{X}(Y_{2})\,,\,Z_{1}\,\rangle
        \,+\,
        \langle\,\i\,\ad_{X}(Y_{2})\,,\,\i\,Z_{2}\,\rangle
    \\
    & = &
        \langle\,\ad_{X}(Y_{1})\,,\,Z_{1}\,\rangle
        \,+\,
        \i\,\langle\,\ad_{X}(Y_{1})\,,\,Z_{2}\,\rangle
        \,+\,
        (-\,\i\,)\,\langle\,\ad_{X}(Y_{2})\,,\,Z_{1}\,\rangle
        \,+\,
        (-\,\i)\,\i\,\langle\,\ad_{X}(Y_{2})\,,\,Z_{2}\,\rangle
    \\
    & = &
        -\,\langle\,Y_{1}\,,\,\ad_{X}(Z_{1})\,\rangle
        \,-\,
        \i\,\langle\,Y_{1}\,,\,\ad_{X}(Z_{2})\,\rangle
        \,-\,
        (-\,\i\,)\,\langle\,Y_{2}\,,\,\ad_{X}(Z_{1})\,\rangle
        \,-\,
        (-\,\i)\,\i\,\langle\,Y_{2}\,,\,\ad_{X}(Z_{2})\,\rangle
    \\
    & = &
        -\,\langle\,Y_{1}\,,\,\ad_{X}(Z_{1})\,\rangle
        \,-\,
        \langle\,Y_{1}\,,\,\ad_{X}(\i\,Z_{2})\,\rangle
        \,-\,
        \langle\,\i\,Y_{2}\,,\,\ad_{X}(Z_{1})\,\rangle
        \,-\,
        \langle\,\i\,Y_{2}\,,\,\ad_{X}(\i\,Z_{2})\,\rangle
    \\
    & = &
        -\,\langle\,Y_{1}\,,\,\ad_{X}(Z_{1}+\i\,Z_{2})\,\rangle
        -\,\langle\,\i\,Y_{2}\,,\,\ad_{X}(Z_{1}+\i\,Z_{2})\,\rangle
    \\
    & = &
        -\,\langle\,Y_{1}+\i\,Y_{2}\,,\,\ad_{X}(Z_{1}+\i\,Z_{2})\,\rangle
    \\
    & = &
        -\,\langle\,Y\,,\,\ad_{X}(Z)\,\rangle\,,
    \quad
    \textnormal{as desired}
    \end{eqnarray*}
\item
    Extension of \,$\ad_{X}$\, by {\color{red}$\C$-linearity} and
    extension of
    \,$\langle\,\cdot\,,\,\cdot\,\rangle$\, by {\color{red}$\C$-bilinearity}
    (anti-linear in the first argument)
    together yield:
    \begin{eqnarray*}
    \langle\,\ad_{X}(Y)\,,\,Z\,\rangle
    & = &
        \langle\,\ad_{X_{1} + \i\,X_{2}}(Y)\,,\,Z\,\rangle
    \;\; = \;\;
        \langle\,\ad_{X_{1}}(Y) + \i\,\ad_{X_{2}}(Y)\,,\,Z\,\rangle
    \\
    & = &
        \langle\,\ad_{X_{1}}(Y)\,,\,Z\,\rangle
        \; + \;
        ({\color{red}-\,\i})\,\langle\,\ad_{X_{2}}(Y)\,,\,Z\,\rangle
    \;\; = \;\;
        {\color{red}-}\,\langle\,Y\,,\,\ad_{X_{1}}(Z)\,\rangle
        \; {\color{red}+} \;
        {\color{red}\i}\,\langle\,Y\,,\,\ad_{X_{2}}(Z)\,\rangle
    \\
    & = &
        \langle\,Y\,,\,\ad_{({\color{red}-}X_{1})}(Z)\,\rangle
        \; + \;
        \langle\,Y\,,\,\ad_{({\color{red}\i}X_{2})}(Z)\,\rangle
    \;\; = \;\;
        \langle\,Y\,,\,\ad_{(-X_{1}+\i X_{2})}(Z)\,\rangle
    \\
    & = &
        \langle\,Y\,,\,\ad_{X^{*}}(Z)\,\rangle\,,
    \end{eqnarray*}
    as required.
    \qed
\end{enumerate}

\vskip 0.5cm
\begin{proposition}[Reductive $\Longrightarrow$ Lie-algebraic orthogonal complements of ideals exist]
\label{ReductiveImpliesOrthogonalComplementsOfIdealsExist}
\mbox{}
\vskip 0.1cm
\noindent
Suppose:
\begin{itemize}
\item
    $\mathfrak{g} \,=\, \mathfrak{k}_{\C}$\,
    is a {\color{red}reductive} complex Lie algebra, equipped with an inner product
    \,$\langle\,\cdot\,,\,\cdot\,\rangle : \mathfrak{g} \times \mathfrak{g} \longrightarrow \C$\,
    as described in Proposition \ref{ReductiveImpliesUnitaryInnerProductOnRealForm}(ii), and
\item
    $\mathfrak{h} \,\subset\, \mathfrak{g}$\, is a Lie-algebra ideal in \,$\mathfrak{g}$.\,
\end{itemize}
Then, the following statements are true:
\begin{enumerate}
\item
    The (inner-product space) orthogonal complement
    \,$\mathfrak{h}^{\perp} \,\subset\, \mathfrak{g}$\,
    of \,$\mathfrak{h}$\, is itself an ideal in \,$\mathfrak{g}$.
\item
    The Lie algebra \,$\mathfrak{g}$\, is equal to the Lie-algebra direct sum of
    \,$\mathfrak{h}$\, and \,$\mathfrak{h}^{\perp}$,\, i.e.,
    \begin{equation*}
    \mathfrak{g}
    \;\; = \;\;
        \mathfrak{h} \,\oplus\, \mathfrak{h}^{\perp}\,,
    \quad
    \textnormal{as Lie algebras}
    \end{equation*}
\end{enumerate}
\end{proposition}
\proof
\begin{enumerate}
\item
    First, recall that, in general:
    \begin{itemize}
    \item
        a linear subspace \,$\,\mathfrak{b} \,\subset\,\mathfrak{a}$\,
        is called a \textbf{subalgebra} of \,$\mathfrak{a}$\, if
        \,$[B_{1},B_{2}] \in \mathfrak{b}$,\, for each \,$B_{1}, B_{2} \in \mathfrak{b}$,
    \item
        a subalgebra \,$\,\mathfrak{b} \,\subset\,\mathfrak{a}$\,
        is called an \textbf{ideal} of \,$\mathfrak{a}$\, if
        \,$\ad_{A}(B) \,:=\, [A,B] \,\in\, \mathfrak{B}$,\, \
        for each \,$A \in \mathfrak{a}$\, and each \,$B \in \mathfrak{b}$;
        thus, an ideal in \,$\mathfrak{a}$\, is exactly the same thing
        as an invariant linear subspace of \,$\mathfrak{a}$\,
        with respect to the adjoint action of \,$\mathfrak{a}$\, on itself
        (Definition 3.5, p.51, \cite{Hall2015}), and
    \item
        the (inner-product-space) orthogonal complement
        \,$\mathfrak{b}^{\perp} \,\subset\, \mathfrak{a}$\,
        of an ideal
        \,$\mathfrak{b} \,\subset\, \mathfrak{a}$\,
        in a real Lie algebra \,$\mathfrak{a}$\, equipped with an inner product
        is itself an ideal in \,$\mathfrak{a}$\,;
        see Proposition 4.27, p.92, \cite{Hall2015}.
    \end{itemize}
    \vskip 0.3cm
    \noindent
    Let
    \,$\ad\vert_{\mathfrak{k}} : \mathfrak{k} \times \mathfrak{k} \longrightarrow \gl(\,\mathfrak{g}\,)$\,
    be the restriction to \,$\mathfrak{k}$\, of the adjoint action \,$\ad$\, of \,$\mathfrak{g}$\, on itself.

    \vskip 0.3cm
    \noindent
    \textbf{Claim 1:}\;\;
    $\mathfrak{h} \,\subset\, \mathfrak{g}$\, is an \,$(\ad\vert_{\mathfrak{k}})$-invariant subspace.
    \vskip 0.05cm
    Proof of Claim 1:\;\; This is immediate; indeed, simply note:
    \begin{equation*}
    \left\{\begin{array}{c}
        \textnormal{$\mathfrak{h}$\, is an}
        \\
        \textnormal{ideal of \,$\mathfrak{g}$}
        \end{array}\right\}
    \;\;\Longleftrightarrow\;\;
    \left\{\begin{array}{c}
        \textnormal{$\mathfrak{h} \,\subset\, \mathfrak{g}$\, is an}
        \\
        \textnormal{$(\ad)$-invariant}
        \\
        \textnormal{subspace}
        \end{array}\right\}
    \;\;\Longrightarrow\;\;
    \left\{\begin{array}{c}
        \textnormal{$\mathfrak{h} \,\subset\, \mathfrak{g}$\, is an}
        \\
        \textnormal{$(\ad\vert_{\mathfrak{k}})$-invariant}
        \\
        \textnormal{subspace}
        \end{array}\right\}
    \end{equation*}
    This proves Claim 1.

    \vskip 0.3cm
    \noindent
    \textbf{Claim 2:}\;\; 
    $\mathfrak{h}^{\perp} \,\subset\, \mathfrak{g}$\,
    is an \,$(\ad\vert_{\mathfrak{k}})$-invariant subspace.
    \vskip 0.05cm
    Proof of Claim 2:\;\;
    This is an immediate consequence of Proposition 4.27, p.92, \cite{Hall2015},
    which states that a ``unitary'' representation of a real Lie algebra
    on a finite-dimensional inner product space is completely reducible.
    This proves Claim 2.

    \vskip 0.3cm
    \noindent
    \textbf{Claim 3:}\;\; 
    $\mathfrak{h}^{\perp} \,\subset\, \mathfrak{g}$\,
    is an \,$(\ad)$-invariant subspace;
    equivalently,
    \,$\mathfrak{h}^{\perp} \,\subset\, \mathfrak{g}$\,
    is an ideal in \,$\mathfrak{g}$.\,
    \vskip 0.05cm
    Proof of Claim 3:\;\;
    Immediate by Claim 2 and $\C$-linearity of
    \,$\ad\vert_{\mathfrak{k}} : \mathfrak{k} \longrightarrow \gl(\,\mathfrak{g}\,)$:
    \begin{equation*}
    \ad_{(X_{1}+\i X_{2})}(Q)
    \; = \;
        \ad_{X_{1}}(Q)
        \, + \,
        \i\,\ad_{X_{2}}(Q)
    \; \in \;
        \mathfrak{h}^{\perp} + \i\,\mathfrak{h}^{\perp}
    \; = \;
        \mathfrak{h}^{\perp}\,,
    \;\;
    \textnormal{for each \,$X_{1},X_{2}\in\mathfrak{k}$,\, $Q \in \mathfrak{h}^{\perp}$}
    \end{equation*}
    This completes the proof of (i).
\item
    We already have the inner-product-space direct sum decomposition:
    \begin{equation*}
    \mathfrak{g}
    \;\; = \;\;
        \mathfrak{h} \,\oplus\, \mathfrak{h}^{\perp}\,,
    \quad
    \textnormal{as inner product spaces}
    \end{equation*}
    Thus, it reamins only to show that the above decomposition
    is also a direct sum of Lie algebras.
    To this end, note that, since both \,$\mathfrak{h}$\, and \,$\mathfrak{h}^{\perp}$\,
    are ideals in \,$\mathfrak{g}$,\, we have:
    \begin{equation*}
    \left.\begin{array}{l}
        X \in \mathfrak{h},
        \\
        Y \in \mathfrak{h}^{\perp}
        \end{array}\quad\right\}
    \!\Longrightarrow\quad
    [X,Y] \;\in\; \mathfrak{h} \cap \mathfrak{h}^{\perp} \;=\; \{\,0\,\} 
    \end{equation*}
    This proves that the direct sum of inner product spaces
    \,$\mathfrak{g} \,=\, \mathfrak{h} \,\oplus\, \mathfrak{h}^{\perp}$\,
    is in fact a direct sum of Lie algebras.
    \vskip 0.3cm
    This completes the proof of (ii), as well as that of the Proposition.
    \qed
\end{enumerate}

\vskip 0.5cm
\begin{theorem}[Semisimple $\Longrightarrow$ direct sum of simple Lie algebras]
\label{SemisimpleImpliesDirectSumOfSimples}
\mbox{}
\vskip 0.1cm
\noindent
If \,$\mathfrak{g}$\, is a non-zero semisimple complex Lie algebra,
then \,$\mathfrak{g}$\, can be expressed as a Lie algebra direct sum
\begin{equation*}
\mathfrak{g}
\;\; = \;\;
    \overset{m}{\underset{j = 1}{\bigoplus}}\;\mathfrak{g}_{j}\,,
\end{equation*}
where each \,$\mathfrak{g}_{j} \subset \mathfrak{g}$\, is a simple Lie algebra.
\end{theorem}
\proof
Either \,$\mathfrak{g}$\, has no non-trivial ideals, or it does.
We will prove that the Theorem is true in either case.

\vskip 0.5cm
\noindent
\underline{\textbf{Case I: \,$\mathfrak{g}$\, has no non-trivial ideals.}}
\vskip 0.2cm
\noindent
Suppose first that \,$\mathfrak{g} \neq \{\,0\,\}$\, has no non-trivial ideals;
equivalently, \,$\mathfrak{g} \neq \{\,0\,\}$\, is irreducible.
We argue that the Theorem is true because, in this case,
\,$\mathfrak{g} \neq \{\,0\,\}$\, is in fact a simple Lie algebra,
which in turn follows immediately from the following general fact:

\vskip 0.3cm
\noindent
\textbf{Claim 1:}\;\;
Every irreducible semisimple Lie algebra is in fact a simple Lie algebra.
\vskip 0.05cm
\noindent
Proof of Claim 1:\;\;
Let \,$\mathfrak{a}$\, be an irreducible semisimple (complex) Lie algebra.
To establish that \,$\mathfrak{a}$\, is in fact a simple Lie algebra,
it suffices only to show that \,$\dim_{\C}(\mathfrak{g}) \,\geq\, 2$.\,
To this end, note that \,$\dim_{\C}(\mathfrak{g}) \,=\, 1$\, would imply that
\,$\mathfrak{g} \neq \{\,0\,\}$\,
is commutative; hence, the centre \,$Z(\mathfrak{g})$\,
of \,$\mathfrak{g}$\, would be \,$\mathfrak{g}$\, itself.
In particular,
\,$Z(\mathfrak{g}) \,=\, \mathfrak{g} \,\neq\, \{\,0\,\}$\,
would be non-trivial, contradicting the semisimplicity hypothesis on \,$\mathfrak{g}$.\,
So, we see that the irreducibility and semisimplicity of \,$\mathfrak{g}$\, together
imply that \,$\dim_{\C}(\mathfrak{g}) \,\geq\, 2$.\,
This proves Claim 1.

\vskip 0.5cm
\noindent
\underline{\textbf{Case II: \,$\mathfrak{g}$\, admits a non-trivial ideal \,$\mathfrak{h} \subset \mathfrak{g}$.}}
\vskip 0.2cm
\noindent
By Proposition \ref{ReductiveImpliesOrthogonalComplementsOfIdealsExist},
we have
\,$\mathfrak{g} \,=\, \mathfrak{h} \oplus \mathfrak{h}^{\perp}$,\,
as Lie algebras (hence, \,$\mathfrak{h}^{\perp}$\, is itself an ideal in \,$\mathfrak{g}$).

\vskip 0.3cm
\noindent
\textbf{Claim 2:}\;\;
If \,$\mathfrak{h}$\, admits a non-trivial ideal \,$\mathfrak{h}^{\prime}$,\,
then \,$\mathfrak{h}^{\prime}$\, is in fact also an ideal in \,$\mathfrak{g}$.\,
\vskip 0.05cm
\noindent
Proof of Claim 2:\;\;
Since \,$\mathfrak{h}^{\prime}$\, is an ideal in \,$\mathfrak{h}$,\, we have:
\begin{equation*}
[H,H^{\prime}]
\; \in \;
    \mathfrak{h}^{\prime}\,,
\quad
\textnormal{for each \,$H \in \mathfrak{h}$\, and each \,$H^{\prime} \in \mathfrak{h}^{\prime}$}
\end{equation*}
On the other hand, we also have:
\begin{equation*}
[Q,H^{\prime}]
\; = \;
    0
\; \in \;
    \mathfrak{h}^{\prime}\,,
\quad
\textnormal{for each \,$Q \in \mathfrak{h}^{\perp}$\, and each \,$H^{\prime} \in \mathfrak{h}^{\prime}$}
\end{equation*}
Bilinearity of the Lie bracket now implies:
\begin{equation*}
[H,H^{\prime}]
\; \in \;
    \mathfrak{h}^{\prime}\,,
\quad
\textnormal{for each \,$H \in \mathfrak{g} = \mathfrak{h} \oplus \mathfrak{h}^{\perp}$\, and each \,$H^{\prime} \in \mathfrak{h}^{\prime}$}
\end{equation*}
This proves Claim 2.

\vskip 0.3cm
\noindent
By Claim 2, we see that
\,$\mathfrak{h}^{\prime\prime} \,:=\, \left(\mathfrak{h}^{\prime}\right)^{\perp} \cap \mathfrak{h}$\,
is an ideal in \,$\mathfrak{g}$,\, and we see that \,$\mathfrak{g}$\,
decomposes as an orthogonal direct sum of Lie algebras, i.e.,
\,$\mathfrak{g} \,=\, \mathfrak{h}^{\prime} \oplus \mathfrak{h}^{\prime\prime} \oplus \mathfrak{h}^{\perp}$.\,
Proceeding analogously, we may conclude that
\,$\mathfrak{g}$\, decomposes into a direct sum of ideals \,$\mathfrak{g}_{j} \subset \mathfrak{g}$,\,
where each \,$\mathfrak{g}_{j}$\, has no non-trival ideals.

\vskip 0.3cm
\noindent
It remains only to show that
\,$\dim_{\C}(\mathfrak{g}_{j}) \,\geq\, 2$,\, for each \,$j$.\,
To this end, note that
\,$\dim_{\C}(\mathfrak{g}_{j}) \,=\, 1$\,
would imply
\,$\mathfrak{g}_{j} \,\neq\, \{\,0\,\}$\, is commutative.
In addition, since elements of \,$\mathfrak{g}_{j}$\,
commute with elements of \,$\mathfrak{g}_{k}$\,
for each \,$k \neq j$,\, we see that
\,$\mathfrak{g}_{j} \,\neq\, \{\,0\,\}$\, must be contained in the centre
\,$Z(\mathfrak{g})$\, of \,$\mathfrak{g}$.\,
This contradicts the semisimplicity hypothesis on \,$\mathfrak{g}$\,
(in particular, the triviality of the centre of \,$\mathfrak{g}$).
Hence, we may now conclude that we indeed have:
\,$\dim_{\C}(\mathfrak{g}_{j}) \,\geq\, 2$,\, for each \,$j$.\,

\vskip 0.3cm
\noindent
So, each \,$\mathfrak{g}_{j}$\, is irreducible and \,$\dim_{\C}(\mathfrak{g}_{j}) \geq 2$,\,
i.e., each \,$\mathfrak{g}_{j}$\, is a simple Lie algebra.
This completes the proof of the Theorem.
\qed
